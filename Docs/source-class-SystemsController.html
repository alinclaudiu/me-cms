 56:     /**
 57:      * Media browser with KCFinder
 58:      */
 59:     public function admin_browser() {
 60:         //Checks for KCFinder
 61:         if(!is_readable(WWW_ROOT.'kcfinder')) {
 62:             $this->Session->flash(__d('me_cms', '%s is not present into %s', 'KCFinder', WWW_ROOT.'kcfinder'), 'error');
 63:             $this->redirect('/admin');
 64:         }
 65:         
 66:         //Checks for uploads directory (`APP/webroot/files`)
 67:         if(!is_writable(WWW_ROOT.'files')) {
 68:             $this->Session->flash(__d('me_cms', 'The directory %s is not readable or writable', WWW_ROOT.'files'), 'error');
 69:             $this->redirect('/admin');
 70:         }
 71:         
 72:         //Gets types
 73:         $types = $this->config['kcfinder']['types'];
 74:         
 75:         //Checks the type, then sets the KCFinder path
 76:         if(!empty($this->request->query['type']) && array_key_exists($this->request->query['type'], $types))
 77:             $this->set('kcfinder', sprintf('/kcfinder/browse.php?lang=%s&type=%s', Configure::read('Config.language'), $this->request->query['type']));
 78:         
 79:         $this->set(array(
 80:             'title_for_layout'  => __d('me_cms', 'Media browser'),
 81:             'types'             => array_combine(array_keys($types), array_keys($types))
 82:         ));
 83:     }

129:             'cakeVersion'       => System::getCakeVersion(),

132:             'imagick'           => System::checkPhpExtension('imagick'),
133:             'logs'              => System::checkLogs(),
134:             'photosWWW'         => System::dirIsWritable(PhotoManager::getFolder()),

137:             'phpVersion'        => System::checkPhpVersion($phpRequired),
138:             'plugins'           => System::getPluginsVersion('MeCms'),
139:             'rewrite'           => System::checkApacheModule('mod_rewrite'),
140:             'tmp'               => System::checkTmp(),
