32: class WidgetHelper extends AppHelper {
33:     /**
34:      * Renders a widget.
35:      * @param array $widget Widget, as array with `name` and (optional) `options` keys
36:      * @return string Html, element
37:      * @throws InternalErrorException
38:      */
39:     public function render($widget) {
40:         if(empty($widget['name']))
41:             throw new InternalErrorException(__d('me_cms', 'Invalid widget name'));
42:         
43:         list($plugin, $name) = pluginSplit($widget['name']);
44:             
45:         $element = empty($plugin) ? sprintf('widgets/%s', $name) : sprintf('%s.widgets/%s', $plugin, $name);
46:         
47:         //Checks if the widget exists
48:         if(!$this->_View->elementExists($element))
49:             throw new InternalErrorException(__d('me_cms', 'The widget %s doesn\'t exist', $widget['name']));
50:             
51:         return $this->_View->element($element, array('options' => empty($widget['options']) ? array() : $widget['options']));
52:     }
53: }
