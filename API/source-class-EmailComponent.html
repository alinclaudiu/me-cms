 49:      * 
 50:      * This is a wrapper for the `CakeEmail` class.
 51:      * @param string $method Method to invoke
 52:      * @param array $params Array of params for the method
 53:      * @uses $cakeEmail
 54:      */
 55:     public function __call($method, $params) {
 56:         return call_user_func_array(array($this->cakeEmail, $method), $params);
 57:     }
 58: 
 59:     /**
 60:      * Construct.
 61:      * @param ComponentCollection $collection A ComponentCollection this component can use to lazy load its components
 62:      * @param array $settings Array of configuration settings.
 63:      * @uses $cakeEmail
 64:      */
 65:     public function __construct(ComponentCollection $collection, $settings = array()) {
 66:         parent::__construct($collection, $settings);
 67:         
 68:         //Initializes the `CakeEmail` class.
 69:         $this->cakeEmail = new CakeEmail();
 70:     }
 71:     
 72:     /**
 73:      * Configures `CakeEmail` with some default settings.
 74:      * @uses $controller
 75:      * @uses set()
 76:      */
 77:     protected function autoConfig() {
 78:         $config = $this->controller->config;
 79:         
 80:         $this->config($config['email']['config']);
 81:         $this->from($from = array($config['email']['webmaster'] => $config['main']['title']));
 82:         $this->sender($from);
 83:         $this->template('default', 'MeCms.default');
 84:         $this->emailFormat('html');
 85:         $this->helpers(array('Html' => array('className' => 'MeTools.MeHtml')));
 86:         $this->set(array(
 87:             'config'        => $config,
 88:             'ip_address'    => $this->controller->request->clientIp(TRUE),
 89:             'site_address'  => Router::url('/', TRUE)
 90:         ));
 91:     }
 92:     
 93:     /**
 94:      * Wrapper for `CakeEmail::viewVars()` method.
 95:      * @param string|array $one A string or an array of data.
 96:      * @param string|array $two Value in case $one is a string (which then works as the key). 
 97:      *  Unused if $one is an associative array, otherwise serves as the values to $one's keys.
 98:      * @return void
 99:      */
100:     public function set($one, $two = NULL) {
101:         if(is_array($one)) {
102:             if(is_array($two))
103:                 return $this->viewVars(array_combine($one, $two));
104:             else
105:                 return $this->viewVars($one);
106:         }
107:         else
108:             return $this->viewVars(array($one => $two));
109:     }
110:

116:     public function startup(Controller $controller) {

120:         $this->autoConfig();
121:     }
