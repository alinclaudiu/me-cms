 43:     public function beforeFilter(\Cake\Event\Event $event) {
 44:         parent::beforeFilter($event);
 45:         
 46:         //Checks if the main folder and its subfolders are writable

 53:             //Gets and sets positions

 56:             //Checks for positions
 57:             if(empty($positions) && !$this->request->isAction('index')) {
 58:                 $this->Flash->alert(__d('me_cms', 'Before you can manage banners, you have to create at least a banner position'));
 59:                 $this->redirect(['controller' => 'BannersPositions', 'action' => 'index']);
 60:             }
 61:         }

 65:     }

 67:     /**
 68:      * Check if the provided user is authorized for the request
 69:      * @param array $user The user to check the authorization of. If empty the user in the session will be used
 70:      * @return bool TRUE if the user is authorized, otherwise FALSE
 71:      * @uses MeCms\Controller\AppController::isAuthorized()

 74:      */
 75:     public function isAuthorized($user = NULL) {
 76:         //Only admins can delete banners
 77:         if($this->request->isAction('delete'))
 78:             return $this->Auth->isGroup('admin');
 79:         
 80:         //Admins and managers can access other actions

 82:     }

 84:     /**
 85:      * Lists banners
 86:      * @uses MeCms\Model\Table\Banners::fromFilter()
 87:      */
 88:     public function index() {
 89:         $this->set('banners', $this->paginate(
 90:             $this->Banners->find()
 91:                 ->contain(['Positions' => ['fields' => ['id', 'name']]])
 92:                 ->select(['id', 'filename', 'target', 'description', 'active', 'click_count'])
 93:                 ->where($this->Banners->fromFilter($this->request->query))
 94:                 ->order(['Banners.filename' => 'ASC'])
 95:         ));
 96:     }
 97:     
 98:     /**
 99:      * Uploads banners
100:      * @uses MeCms\Controller\_upload()
101:      * @uses MeCms\Utility\BannerFile::folder()
102:      */
103:     public function upload() {
104:         //If there's only one position, it automatically sets the query value
105:         if(!$this->request->query('position') && count($this->viewVars['positions']) < 2)
106:             $this->request->query['position'] = fk($this->viewVars['positions']);
107:         
108:         $position = $this->request->query('position');
109:         
110:         if($position && $this->request->data('file'))
111:             //Checks if the file has been uploaded
112:             if($filename = $this->_upload($this->request->data('file'), BannerFile::folder()))
113:                 $this->Banners->save($this->Banners->newEntity([
114:                     'position_id'   => $position,
115:                     'filename'      => basename($filename)
116:                 ]));
117:     }

124:     public function edit($id = NULL)  {
125:         $banner = $this->Banners->get($id);
126:         
127:         if($this->request->is(['patch', 'post', 'put'])) {
128:             $banner = $this->Banners->patchEntity($banner, $this->request->data);
129:

131:                 $this->Flash->success(__d('me_cms', 'The banner has been saved'));
132:                 return $this->redirect(['action' => 'index']);
133:             }

135:                 $this->Flash->error(__d('me_cms', 'The banner could not be saved'));
136:         }
