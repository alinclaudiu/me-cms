 72:         //Loads the KcFinder component
 73:         $this->loadComponent('MeCms.KcFinder');
 74:         
 75:         //Checks for KCFinder
 76:         if(!$this->KcFinder->checkKcfinder())
 77:             throw new InternalErrorException(__d('me_cms', '{0} is not present into {1}', 'KCFinder', rtr($this->KcFinder->getKcfinderPath())));

 79:         //Checks for the files directory (`APP/webroot/files`)
 80:         if(!$this->KcFinder->checkFiles())
 81:             throw new InternalErrorException(__d('me_tools', 'File or directory `{0}` not writeable', rtr($this->KcFinder->getFilesPath())));
 82:         
 83:         //Gets the supperted types from configuration
 84:         $types = $this->KcFinder->getTypes();
 85:         
 86:         //If there's only one type, it automatically sets the query value
 87:         if(!$this->request->query('type') && count($types) < 2)
 88:             $this->request->query['type'] = fk($types);
 89:         
 90:         //Gets the type from the query and the types from configuration
 91:         $type = $this->request->query('type');
 92:         
 93:         $locale = substr(\Cake\I18n\I18n::locale(), 0, 2);
 94:         
 95:         //Checks the type, then sets the KCFinder path
 96:         if($type && array_key_exists($type, $types))
 97:             $this->set('kcfinder', sprintf('%s/kcfinder/browse.php?lang=%s&type=%s', Router::url('/vendor', TRUE), empty($locale) ? 'en' : $locale, $type));
 98:         
 99:         $this->set('types', array_combine(array_keys($types), array_keys($types)));
100:     }

202:     /**
203:      * Logs viewer
204:      */
205:     public function logs_viewer() {
206:         //Gets all log files
207:         $files = (new Folder(LOGS))->read(TRUE, ['empty'])[1];
208:                 
209:         //The array keys will be the filename without extension
210:         $files = array_combine(array_map(function($v) { return pathinfo($v, PATHINFO_FILENAME); }, $files), $files);
211:         
212:         //If there's only one log file, it automatically sets the query value
213:         if(!$this->request->query('file') && count($files) < 2)
214:             $this->request->query['file'] = fk($files);
215:         
216:         //If a log file has been specified
217:         if($this->request->query('file') && $this->request->is('get')) {
218:             //Gets the log content
219:             $logs = file_get_contents(LOGS.sprintf('%s.log', $this->request->query('file')));
220:             
221:             //Tries to unserialized
222:             $unserialized = @unserialize($logs);
223:             
224:             if($unserialized !== FALSE) {
225:                 $this->set('unserialized_logs', $unserialized);
226:             }
227:             else
228:                 $this->set('plain_logs', $logs);
229:         }
230:         
231:         $this->set(compact('files'));
232:     }

234:     /**
235:      * Temporary cleaner (assets, cache, logs and thumbnails)
236:      * @param string $type Type
237:      * @uses MeTools\Cache\Cache::clearAll()
238:      */
239:     public function tmp_cleaner($type) {
240:         if(!$this->request->is(['post', 'delete']))
241:             return $this->redirect(['action' => 'tmp_viewer']);
242:         
243:         switch($type) {
244:             case 'all':
245:                 $success = clear_dir(ASSETS) && clear_dir(LOGS) && Cache::clearAll() && clear_dir(THUMBS);
246:                 break;
247:             case 'cache':
248:                 $success = Cache::clearAll();
249:                 break;
250:             case 'assets':
251:                 $success = clear_dir(ASSETS);
252:                 break;
253:             case 'logs':
254:                 $success = clear_dir(LOGS);
255:                 break;
256:             case 'thumbs':
257:                 $success = clear_dir(THUMBS);
258:                 break;
259:         }
260:         
261:         if(!empty($success))
262:             $this->Flash->success(__d('me_cms', 'The operation has been performed correctly'));
263:         else
264:             $this->Flash->error(__d('me_cms', 'The operation has not been performed correctly'));
265:         
266:         return $this->redirect(['action' => 'tmp_viewer']);
267:     }
